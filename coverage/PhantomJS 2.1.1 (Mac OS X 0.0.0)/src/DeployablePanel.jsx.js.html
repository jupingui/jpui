define('jpui/DeployablePanel.jsx', ['require', 'react', './mixins/InputWidget', './core/componentTools', './core/language', './core/util'], function (require) {
    var React = require('react');
    var InputWidget = require('./mixins/InputWidget');
    var cTools = require('./core/componentTools');
    var language = require('./core/language').deployablePanel;
    var util = require('./core/util');
    var labels = {
        expand: language.expand,
        hide: language.hide
    };
    return React.createClass({
        mixins: [InputWidget],
        getDefaultProps: function getDefaultProps() {
            return {
                skin: '',
                className: '',
                style: {},
                disabled: false,
                labels: labels,
                switchLocation: 'top',
                valueTemplate: ''
            };
        },
        getInitialState: function getInitialState() {
            return {};
        },
        onChange: function onChange(e) {
            e = { target: this.refs.container };
            e.target.value = this.___getValue___() === 'expand' ? 'hide' : 'expand';
            this.___dispatchChange___(e);
        },
        render: function render() {
            var containerProp = cTools.containerBaseProps('deployablepanel', this);
            var value = this.___getValue___();
            var labels = util.extend({}, labels, this.props.labels);
            return React.createElement('div', containerProp, this.props.switchLocation === 'top' ? React.createElement('div', { className: 'opt-bar' }, React.createElement('span', {
                onClick: this.onChange,
                className: 'opt-link'
            }, value === 'expand' ? labels.hide : labels.expand), React.createElement('span', {
                onClick: this.onChange,
                className: 'jpui-icon jpui-icon-arrow-' + (value === 'expand' ? 'up' : 'down')
            })) : null, React.createElement('div', {
                className: 'content-bar',
                style: { display: value === 'expand' ? 'block' : 'none' }
            }, this.props.children), this.props.switchLocation !== 'top' ? React.createElement('div', { className: 'opt-bar' }, React.createElement('span', {
                onClick: this.onChange,
                className: 'opt-link'
            }, value === 'expand' ? labels.hide : labels.expand), React.createElement('span', {
                onClick: this.onChange,
                className: 'jpui-icon jpui-icon-arrow-' + (value === 'expand' ? 'up' : 'down')
            })) : null);
        }
    });
});
