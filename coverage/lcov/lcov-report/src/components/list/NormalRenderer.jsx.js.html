define('jpui/components/list/NormalRenderer.jsx', ['require', 'react'], function (require) {
    var React = require('react');
    return React.createClass({
        getDefaultProps: function getDefaultProps() {
            return {
                label: '',
                value: '',
                disabled: false,
                onClick: function onClick() {}
            };
        },
        getInitialState: function getInitialState() {
            return {};
        },
        onClick: function onClick(e) {
            if (this.props.disabled) return;
            e = { target: this.refs.container };
            e.target.value = this.props.value;
            this.props.onClick(e);
        },
        render: function render() {
            var containerProp = {
                ref: 'container',
                className: 'list-normal-item' + (this.props.disabled ? ' list-normal-item-disabled' : ''),
                onClick: this.onClick
            };
            return React.createElement('div', containerProp, React.createElement('span', null, this.props.label));
        }
    });
});
