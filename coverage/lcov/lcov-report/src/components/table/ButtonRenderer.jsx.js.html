define('jpui/components/table/ButtonRenderer.jsx', ['require', 'react'], function (require) {
    var React = require('react');
    return React.createClass({
        getDefaultProps: function getDefaultProps() {
            return {
                className: '',
                style: {},
                item: {},
                row: -1,
                column: -1,
                onAction: function onAction() {}
            };
        },
        onIconClick: function onIconClick(e) {
            this.props.onAction('ButtonRendererClick', {
                item: this.props.item,
                row: this.props.row,
                column: this.props.column,
                command: this.props.command
            });
        },
        render: function render() {
            var item = this.props.item;
            var value = this.props.content;
            var tdProp = {
                className: 'td-button ' + this.props.className,
                style: this.props.style,
                title: this.props.title
            };
            var buttonIcon = this.props.buttonIcon;
            var iconProp = {
                className: 'is-icon font-icon' + (typeof buttonIcon === 'string' ? ' ' + buttonIcon : ''),
                style: {
                    float: tdProp.style.textAlign === 'right' ? 'left' : 'right',
                    marginTop: 2
                },
                onClick: this.onIconClick
            };
            var labelProp = {
                className: 'btn-label',
                style: { float: tdProp.style.textAlign === 'right' ? 'left' : 'right' },
                onClick: this.onIconClick
            };
            var hideButton = this.props.item['hideButton-' + this.props.field];
            return React.createElement('td', tdProp, !hideButton ? React.createElement('div', iconProp) : null, this.props.buttonLabel && !hideButton ? React.createElement('div', labelProp, this.props.buttonLabel) : null, React.createElement('div', {
                style: {
                    margin: '10px 0 5px',
                    lineHeight: '20px'
                }
            }, value));
        }
    });
});
