define('jpui/components/tree/NormalRenderer.jsx', ['require', 'react', '../../mixins/TreeNodeBase'], function (require) {
    var React = require('react');
    var TreeNodeBase = require('../../mixins/TreeNodeBase');
    return React.createClass({
        mixins: [TreeNodeBase],
        getDefaultProps: function getDefaultProps() {
            return {
                index: '',
                item: {},
                value: {},
                treeComponent: {},
                disabled: false,
                onAction: function onAction() {}
            };
        },
        getInitialState: function getInitialState() {
            return {};
        },
        onClick: function onClick(e) {
            if (this.props.disabled) return;
            typeof this.props.onAction === 'function' && this.props.onAction('TreeLeafClick', {
                item: this.props.item,
                index: this.props.index.split(',')
            });
        },
        render: function render() {
            var item = this.props.item;
            var expand = this.props.value.expand || {};
            var containerProp = {
                ref: 'container',
                className: 'list-normal-item' + (this.props.disabled ? ' list-normal-item-disabled' : ''),
                onClick: this.onClick,
                style: { paddingLeft: (this.props.index.split(',').length - 1) * 12 }
            };
            var iconProp = {
                className: 'expand-button jpui-icon jpui-icon-arrow-' + (expand[item.value] ? 'down' : 'right'),
                style: { visibility: !(item.children instanceof Array) ? 'hidden' : 'visible' },
                onClick: this.onExpand
            };
            return React.createElement('div', containerProp, React.createElement('div', iconProp), React.createElement('span', null, item.label));
        }
    });
});
