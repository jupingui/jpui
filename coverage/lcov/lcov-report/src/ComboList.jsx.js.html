define('jpui/ComboList.jsx', ['require', 'react', './Layer.jsx', './List.jsx', './core/componentTools'], function (require) {
    var React = require('react');
    var Layer = require('./Layer.jsx');
    var List = require('./List.jsx');
    var cTools = require('./core/componentTools');
    return React.createClass({
        contextTypes: { appSkin: React.PropTypes.string },
        getDefaultProps: function getDefaultProps() {
            return {
                skin: '',
                className: '',
                style: {},
                disabled: false,
                label: 'ComboList',
                icon: '',
                value: '',
                datasource: [],
                onClick: function onClick() {}
            };
        },
        getInitialState: function getInitialState() {
            return { layerOpen: false };
        },
        onListClick: function onListClick(e) {
            if (this.props.disabled) return;
            this.props.onClick(e);
            this.setState({ layerOpen: false });
        },
        onMainButtonClick: function onMainButtonClick(e) {
            if (this.props.disabled) return;
            e = { target: this.refs.container };
            e.target.value = this.props.value;
            this.setState({ layerOpen: false });
            this.props.onClick(e);
        },
        onDropDownButtonClick: function onDropDownButtonClick(e) {
            if (this.props.disabled) return;
            this.setState({ layerOpen: true });
            e.stopPropagation();
        },
        onMouseLeave: function onMouseLeave(e) {
            if (this.props.disabled) return;
            cTools.closeLayerHandlerFactory(this, 'layerOpen')();
        },
        render: function render() {
            var me = this;
            var containerProp = cTools.containerBaseProps('combolist', this, {
                merge: {
                    onMouseLeave: this.onMouseLeave,
                    onClick: this.onMainButtonClick
                },
                widthCorrect: -2
            });
            var hasDatasource = this.props.datasource instanceof Array && this.props.datasource.length > 0;
            var dropdownButtonProp = {
                className: 'icon-right jpui-icon jpui-icon-arrow-down' + (this.state.layerOpen && hasDatasource && !this.props.disabled ? ' layerOpen' : ''),
                onClick: this.onDropDownButtonClick
            };
            var labelProp = { className: 'label-container' + (hasDatasource ? ' marginRight' : '') + (this.props.icon ? ' font-icon ' + this.props.icon : '') };
            var layerProp = {
                ref: 'layer',
                isOpen: this.state.layerOpen && hasDatasource && !this.props.disabled,
                anchor: this.refs.container,
                onMouseLeave: this.onMouseLeave,
                skin: this.context.appSkin ? this.context.appSkin + '-important' : 'normal'
            };
            var listProp = {
                datasource: this.props.datasource,
                ref: 'list',
                skin: 'important',
                onClick: this.onListClick,
                style: {
                    maxHeight: '242px',
                    overflowY: 'auto',
                    overflowX: 'hidden'
                }
            };
            return React.createElement('div', containerProp, hasDatasource ? React.createElement('div', dropdownButtonProp) : null, hasDatasource ? React.createElement('div', { className: 'spliter' }) : null, React.createElement('span', labelProp, this.props.label), React.createElement(Layer, layerProp, React.createElement(List, listProp)));
        }
    });
});
