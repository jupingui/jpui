define('jpui/Calendar.jsx', ['require', 'react', './core/language', './core/util', './core/calendarTools', './core/componentTools', './factories/calendarFactory.jsx', './mixins/InputWidget'], function (require) {
    var React = require('react');
    var language = require('./core/language');
    var util = require('./core/util');
    var tool = require('./core/calendarTools');
    var cTools = require('./core/componentTools');
    var factory = require('./factories/calendarFactory.jsx');
    var InputWidget = require('./mixins/InputWidget');
    return React.createClass({
        contextTypes: { appSkin: React.PropTypes.string },
        childContextTypes: { appSkin: React.PropTypes.string },
        getChildContext: function getChildContext() {
            return { appSkin: '' };
        },
        mixins: [InputWidget],
        getDefaultProps: function getDefaultProps() {
            return {
                skin: '',
                className: '',
                style: {},
                disabled: false,
                min: '0-1-1',
                max: '9999-12-31',
                valueTemplate: util.dateFormat(null, 'YYYY-MM-DD')
            };
        },
        getInitialState: function getInitialState() {
            var value = tool.str2date(this.___getValue___()) || new Date();
            return {
                displayYear: value.getFullYear(),
                displayMonth: value.getMonth(),
                inputYear: value.getFullYear(),
                inputMonth: value.getMonth() + 1,
                inRange: true
            };
        },
        onDayClick: function onDayClick(e) {
            if (this.props.disabled) return;
            var timer = tool.str2date(this.state.displayYear + '-' + (this.state.displayMonth + 1) + '-' + e.target.value);
            e = { target: this.refs.container };
            e.target.value = util.dateFormat(timer, 'YYYY-MM-DD');
            this.___dispatchChange___(e);
        },
        onYearChange: function onYearChange(e) {
            if (this.props.disabled) return;
            var year = this.state.displayYear;
            if (!isNaN(e.target.value)) {
                year = e.target.value * 1;
            }
            this.setState({
                inputYear: e.target.value,
                displayYear: year,
                inRange: tool.monthInRange(year, this.state.displayMonth, this.props.min, this.props.max)
            });
        },
        onMonthChange: function onMonthChange(e) {
            if (this.props.disabled) return;
            var month = this.state.displayMonth;
            if (!isNaN(e.target.value)) {
                month = e.target.value * 1 - 1;
                month = month < -1 ? -1 : month;
                month = month > 11 ? 11 : month;
            }
            this.setState({
                inRange: tool.monthInRange(this.state.displayYear, month, this.props.min, this.props.max),
                inputMonth: e.target.value,
                displayMonth: month
            });
        },
        onMonthAdd: function onMonthAdd(e) {
            if (this.props.disabled) return;
            var month = this.state.displayMonth;
            var year = this.state.displayYear;
            month = month * 1 + 1;
            if (month > 11) {
                month = 0;
                year = year * 1 + 1;
            }
            this.setState({
                inRange: tool.monthInRange(year, month, this.props.min, this.props.max),
                displayYear: year,
                displayMonth: month,
                inputYear: year,
                inputMonth: month + 1
            });
        },
        onMonthSub: function onMonthSub(e) {
            if (this.props.disabled) return;
            var month = this.state.displayMonth;
            var year = this.state.displayYear;
            month = month * 1 - 1;
            if (month < 0) {
                month = 11;
                year = year * 1 - 1;
            }
            this.setState({
                inRange: tool.monthInRange(year, month, this.props.min, this.props.max),
                displayYear: year,
                displayMonth: month,
                inputYear: year,
                inputMonth: month + 1
            });
        },
        render: function render() {
            return React.createElement('div', cTools.containerBaseProps('calendar', this), this.context.appSkin !== 'oneux4' ? factory.normalOperationBarFactory(this) : factory.oneux4OperationBarFactory(this), React.createElement('div', { className: 'calendar-day-label' }, this.state.inRange ? language.calendar.day.map(factory.dayLabelFactory) : this.props.min.replace(/-/g, '.') + ' - ' + this.props.max.replace(/-/g, '.')), React.createElement('div', { className: 'calendar-buttons' }, factory.buttonFactory(this)));
        }
    });
});
